\newcommand{\hsrdesc}{High-availability Seamless Redundancy. Redundanzprotokoll für Ethernet basierte Netzwerke. HSR ist für redundant gekoppelte Ringtopologien ausgelegt. Die Datenübermittlung innerhalb eines HSR-Rings ist im Fehlerfall gewährleistet, wenn eine Netzwerkschnittstelle ausfallen sollte.}

\newcommand{\ietdesc}{Interspersed Express Traffic. Erlaubt es Frames während des Sendevorgangs zu unterbrechen, um sogenannte Express Frames so schnell wie möglich versenden zu können.}


% *****************************************************************************
% Nomenclature entries

\nomenclature{HSR}{\hsrdesc}
\nomenclature{IET}{\ietdesc}
